//c++ code for sorting
#include<bits/stdc++.h>
 
using namespace std;
 
int main()
{
    int a[50],n,i,j,temp;
    cout<<"Enter the size of array: ";
    cin>>n;
    cout<<"Enter the array elements: "; 
    // no need to tell that
    for(i=0;i<n;++i)
        cin>>a[i];
    //yeah we know that
    for(i=1;i<n;++i)
    {
        for(j=0;j<(n-i);++j)
            if(a[j]>a[j+1])
            {
                temp=a[j];
                a[j]=a[j+1];
                a[j+1]=temp;
            }
    }
    
    cout<<"Array after the bubble sort: ";
    for(i=0;i<n;++i)
        cout<<" "<<a[i]<<" ";
        
    return 0;
}
