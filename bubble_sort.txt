//c++ code for sorting
#include<bits/stdc++.h>
 
using namespace std;
 
int main()
{
    int n,i,j,temp;
    cout<<"Enter the size of array: ";
    cin>>n;
    int a[n];
    cout<<"Enter the array elements: "; 
    // i has range from i to size of array -1
    for(i=0;i<n;++i)
        cin>>a[i];
    //For Loop after Input.
    for(i=0;i<n-1;++i)
    {
        for(j=0;j<(n-i-1);++j)
            if(a[j]>a[j+1])
            {
                temp=a[j];
                a[j]=a[j+1];
                a[j+1]=temp;
            }
    }
    
    cout<<"Array after the bubble sort: ";
    for(i=0;i<n;++i)
        cout<<" "<<a[i]<<" ";
        
    return 0;
}
